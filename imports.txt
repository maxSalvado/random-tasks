import { Routes } from '@angular/router';

// Lazy loading (recommended for standalone)
export const routes: Routes = [
  { path: 'login', loadComponent: () => import('./pages/login/login.component').then(m => m.LoginComponent) },
  { path: 'home', loadComponent: () => import('./pages/home/home.component').then(m => m.HomeComponent) },
  {
    path: 'automation',
    children: [
      { path: '', loadComponent: () => import('./pages/automation/automation.component').then(m => m.AutomationComponent) },
      { path: ':name', loadComponent: () => import('./pages/automation/automation-detail.component').then(m => m.AutomationDetailComponent) },
    ],
  },
  { path: 'experiencia-em-clientes', loadComponent: () => import('./pages/experiencia-em-clientes/experiencia.component').then(m => m.ExperienciaComponent) },
  {
    path: 'tecnologias-avancadas',
    children: [
      { path: '', loadComponent: () => import('./pages/tecnologias-avancadas/tech.component').then(m => m.TechComponent) },
      { path: ':name', loadComponent: () => import('./pages/tecnologias-avancadas/tech-detail.component').then(m => m.TechDetailComponent) },
    ],
  },
  {
    path: 'oportunidades',
    children: [
      { path: '', loadComponent: () => import('./pages/oportunidades/oportunidades.component').then(m => m.OportunidadesComponent) },
      { path: 'criar', loadComponent: () => import('./pages/oportunidades/criar.component').then(m => m.CriarOportunidadeComponent) },
      { path: 'visualizar/:id', loadComponent: () => import('./pages/oportunidades/visualizar.component').then(m => m.VisualizarOportunidadeComponent) },
      { path: 'editar/:id', loadComponent: () => import('./pages/oportunidades/editar.component').then(m => m.EditarOportunidadeComponent) },
    ],
  },
  {
    path: 'pocs',
    children: [
      { path: '', loadComponent: () => import('./pages/pocs/pocs.component').then(m => m.PocsComponent) },
      { path: 'criar', loadComponent: () => import('./pages/pocs/criar.component').then(m => m.CriarPocComponent) },
      { path: 'visualizar/:id', loadComponent: () => import('./pages/pocs/visualizar.component').then(m => m.VisualizarPocComponent) },
      { path: 'editar/:id', loadComponent: () => import('./pages/pocs/editar.component').then(m => m.EditarPocComponent) },
    ],
  },
  { path: '', redirectTo: 'login', pathMatch: 'full' },
  { path: '**', redirectTo: 'login' },
];
