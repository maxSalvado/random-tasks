import { Component, computed, inject } from '@angular/core';
import { IonicModule } from '@ionic/angular';
import { ActivatedRoute, ParamMap } from '@angular/router';
import { CommonModule } from '@angular/common';
import { toSignal } from '@angular/core/rxjs-interop';

@Component({
  selector: 'app-visualizar-oportunidade',
  standalone: true,
  imports: [IonicModule, CommonModule],
  template: `
    <ion-header>
      <ion-toolbar>
        <ion-title>Visualizar Oportunidade</ion-title>
      </ion-toolbar>
    </ion-header>

    <ion-content class="ion-padding">
      <p>ID da oportunidade: {{ id() }}</p>
    </ion-content>
  `,
})
export class VisualizarOportunidadeComponent {
  private route = inject(ActivatedRoute);
  private idSignal = toSignal(this.route.paramMap.pipe(
    // paramMap is Observable<ParamMap>
    // `get('id')` extracts the value from URL
    map((params: ParamMap) => params.get('id') ?? '')
  ));

  id = computed(() => this.idSignal());
}
